
reservation table:
id (integer), room_id (integer), customer_id (integer), check_in_date (date), check_out_date (date), num_of_pers(integer)



room table:
id (integer), room_type_id (integer), room_num (integer), room_details_id(integer), property_id (integer)

room_details_id table:
id (integer), smoking (boolean), wifi (boolean), AC(boolean), description (string)

room_type table:
id (integer), type (string), max_pers (integer), images (string)



property table:
id (integer), name (string), type_id (integer), location_id (integer), property_details_id (integer).

property_details_id table:
id (integer), parking (boolean), restaurant (boolean), swimming_pool (boolean), pets_allowed (boolean)

property_category table:
id (integer), category (string)



location table:
id (integer), country_id (integer) city_id (integer), zipcode (string) street_number (string).

country table:
id (integer), country_name(string)

city table:
id (integer), city_name(string)


customer table:
id (integer), first_name (string), last_name (string), email (string)
